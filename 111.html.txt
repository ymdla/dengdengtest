        function computeScores() {
            const sums = new Array(15).fill(0);
            const counts = new Array(15).fill(0);
            for (let qIdx=0; qIdx<39; qIdx++) {
                const ans = answers[qIdx];
                if (!ans) continue;
                const scoreMap = OPTION_SCORES[qIdx+1]?.[ans];
                if (!scoreMap) continue;
                Object.entries(scoreMap).forEach(([dimStr, val]) => {
                    const dim = parseInt(dimStr)-1;
                    sums[dim] += val;
                    counts[dim] += 1;
                });
            }
            const avg = new Array(15).fill(0);
            for (let i=0; i<15; i++) avg[i] = counts[i] > 0 ? sums[i] / counts[i] : 0;
            return avg;
        }

        function cosineSim(a,b) {
            let dot=0, normA=0, normB=0;
            for(let i=0; i<a.length; i++) {
                dot += a[i]*b[i];
                normA += a[i]*a[i];
                normB += b[i]*b[i];
            }
            return (normA===0 || normB===0) ? 0 : dot/(Math.sqrt(normA)*Math.sqrt(normB));
        }

        function matchType(scores) {
            const sims = ARCHETYPES.map(vec => cosineSim(scores, vec));
            let bestIdx=0, bestSim=sims[0];
            sims.forEach((s,i)=>{ if(s>bestSim){ bestSim=s; bestIdx=i; } });
            let secondSim = -1, secondIdx = -1;
            sims.forEach((s,i)=>{ if(i!==bestIdx && s>secondSim){ secondSim=s; secondIdx=i; } });
            if (bestSim >= 0.75) {
                if (secondSim >= 0.65 && (bestSim - secondSim) < 0.1)
                    return { bestIdx, bestSim, secondIdx, secondSim, mixed: true, confidence: 'mixed' };
                return { bestIdx, bestSim, mixed: false, confidence: 'strong' };
            } else if (bestSim >= 0.60) {
                return { bestIdx, bestSim, mixed: false, confidence: 'weak' };
            } else {
                return { bestIdx: -1, bestSim, mixed: false, confidence: 'none' };
            }
        }

        function normalizeScore(v) { return (v + 2) * 25; }

        function detectConflicts(scores) {
            const normScores = scores.map(v => normalizeScore(v));
            const conflicts = [];
            CONFLICT_PAIRS.forEach(pair => {
                const valA = normScores[pair.dimA];
                const valB = normScores[pair.dimB];
                const condA = pair.dirA === 'high' ? valA > 62 : valA < 38;
                const condB = pair.dirB === 'high' ? valB > 62 : valB < 38;
                if (condA && condB) {
                    const deviation = Math.abs(valA - 50) + Math.abs(valB - 50);
                    conflicts.push({ ...pair, deviation, valA, valB });
                }
            });
            conflicts.sort((a,b) => b.deviation - a.deviation);
            return conflicts;
        }

        function findClosestTwoArchetypes(scores) {
            const sims = ARCHETYPES.map((vec, idx) => ({ idx, sim: cosineSim(scores, vec) }));
            sims.sort((a,b) => b.sim - a.sim);
            return [sims[0].idx, sims[1].idx];
        }
